i = 0;
j;
max = 0;
soma = 0;
med;
x;
n = 5;
m = 5;
matrix [n][m]; 

START

WHILE (i < n) {
	
	j = 0;	

	WHILE (j < n) {

		PRINT ("Insert next number: ");
		READ (x);

		matrix [i][j] = x;

		soma = soma + x;

		IF (x > max) {

			max = x;

		}

		j = j + 1;

	}

	i = i + 1;

}

med = soma / (n * m);

PRINT ("Average = ");
PRINT (med);

PRINT ("Max = ");
PRINT (max);

END
