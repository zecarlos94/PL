i; j; x; n; pos; swap; array [5];

START

WHILE (i < n) {

	PRINT ("Insert number: ");
	READ (x);

	array [i] = x;

	i = i + 1;
}

i = 0;

WHILE (i < n) {

	pos = i;

	j = i + 1;

	WHILE (j < n) {

		IF (array [pos] > array [j]) {

			pos = j;

		}
	}

	IF (pos != i) {

		swap = array [i];
		array [i] = array [pos];
		array [pos] = swap;

	}
}

i = 0;

WHILE (i < n) {

	PRINT (array [i]);

	IF (i != n - 1) {

		PRINT (" ,");

	}

	i = i + 1;
}

END
